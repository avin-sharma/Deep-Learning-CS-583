PCA and the standard autoencoder are unsupervised dimensionality reduction methods, and their learned features are not discriminative. If you build a classifier upon the low-dimenional features extracted by PCA and autoencoder, you will find the classification accuracy very poor.

Linear discriminant analysis (LDA) is a traditionally supervised dimensionality reduction method for learning low-dimensional features which are highly discriminative. Likewise, can we extend autoencoder to supervised leanring?

import numpy

def to_one_hot(y, num_class=10):
    results = numpy.zeros((len(y), num_class))
    for i, label in enumerate(y):
        results[i, label] = 1.
    return results

from keras.layers import Dense, Input, Activation, Dropout, BatchNormalization
from keras import models

input_img = Input(shape=(784,), name='input_img')

encode1 = Dense(128, name='encode1')(input_img)
bn1 = BatchNormalization()(encode1)
activation1 = Activation('relu', name = 'activation1')(bn1)

dropout1 = Dropout(0.2)(activation1)
encode2 = Dense(32, activation='relu', name='encode2')(dropout1)
bn4 = BatchNormalization()(encode2)
activation2 = Activation('relu', name = 'activation2')(bn4)

encode3 = Dense(8, activation='relu', name='encode3')(activation2)
bn5 = BatchNormalization()(encode3)
activation3 = Activation('relu', name = 'activation3')(bn5)

encode4 = Dense(8, activation='relu', name='encode4')(activation3)
activation8 = Activation('relu', name = 'activation8')(encode4)

bottleneck = Dense(2, activation='relu', name='bottleneck')(activation8)

decode1 = Dense(8, activation='relu', name='decode1')(bottleneck)
activation4 = Activation('relu', name = 'activation4')(decode1)

decode2 = Dense(32, activation='relu', name='decode2')(activation4)
bn3 = BatchNormalization()(decode2)
activation5 = Activation('relu', name = 'activation5')(bn3)

decode3 = Dense(128, activation='relu', name='decode3')(activation5)
bn6 = BatchNormalization()(decode3)
activation6 = Activation('relu', name = 'activation6')(bn6)

dropout2 = Dropout(0.3)(activation6)
decode4 = Dense(784, activation='relu', name='decode4')(dropout2)
activation7 = Activation('relu', name = 'activation7')(dropout2)

ae = models.Model(input_img, decode4)

# build the supervised autoencoder network
from keras.layers import Dense, Input
from keras import models

input_img = Input(shape=(784,), name='input_img')

# encoder network

encode1 = Dense(128, name='encode1')(input_img)
bn1 = BatchNormalization()(encode1)
activation1 = Activation('relu')(bn1)
dp1 = Dropout(0.1)(activation1)

encode2 = Dense(32, name='encode2')(dp1)
bn2 = BatchNormalization()(encode2)
activation2 = Activation('relu')(bn2)

encode3 = Dense(8, name='encode3')(activation2)
bn9 = BatchNormalization()(encode3)
activation9 = Activation('relu')(bn9)
# The width of the bottleneck layer must be exactly 2.
bottleneck = Dense(2, activation='relu', name='bottleneck')(activation9)

decode1 = Dense(8, name='decode1')(bottleneck)
bn3 = BatchNormalization()(decode1)
activation3 = Activation('relu')(bn3)

decode2 = Dense(32, name='decode2')(activation3)
bn4 = BatchNormalization()(decode2)
activation4 = Activation('relu')(bn4)

decode3 = Dense(128, name='decode3')(activation4)
bn8 = BatchNormalization()(decode3)
activation8 = Activation('relu')(bn8)
dp2 = Dropout(0.2)(activation8)

decode4 = Dense(784, activation='relu', name='decode4')(dp2)


# build a classifier upon the bottleneck layer
classifier1 = Dense(8, name='classifier1')(bottleneck)
bn5 = BatchNormalization()(classifier1)
activation5 = Activation('relu')(bn5)

classifier2 = Dense(32, name='classifier2')(activation5)
bn6 = BatchNormalization()(classifier2)
activation6 = Activation('relu')(bn6)

classifier3 = Dense(128, name='classifier3')(activation6)
bn7 = BatchNormalization()(classifier3)
activation7 = Activation('relu')(bn7)

dp3 = Dropout(0.2)(activation7)
classifier4 = Dense(10, activation='softmax', name='classifier4')(dp3)
